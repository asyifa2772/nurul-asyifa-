    <script>
        // Fungsi saat tombol "Yes" diklik
        function showResult() {
            document.getElementById('resultText').classList.add('show');
        }

        // Trik usil: Membuat tombol "No" menghindar saat didekati/diklik
        function moveButton() {
            const btnNo = document.getElementById('btnNo');
            // Menghitung posisi acak agar tombol tidak keluar dari layar
            const x = Math.random() * (window.innerWidth - btnNo.offsetWidth - 20);
            const y = Math.random() * (window.innerHeight - btnNo.offsetHeight - 20);
            
            btnNo.style.position = 'absolute';
            btnNo.style.left = x + 'px';
            btnNo.style.top = y + 'px';
        }
    </script>
